create database Nauka_Biblioteka_Alana DEFAULT CHARACTER SET utf8 COLLATE utf8_polish_ci;
use Nauka_Biblioteka_Alana;
create table Autorzy
(
Nr_a int primary key AUTO_INCREMENT,
Imie varchar(25),
Nazwisko varchar(25)
);

create table Ksiazki
(
Nr_ks int primary key AUTO_INCREMENT,
Tytul varchar(60),
Rok int,
Nr_a int,
FOREIGN KEY (Nr_a) references Autorzy (Nr_a),
Gatunek varchar(25),
Cena FLOAT
);

create table Klienci
(
Nr_kl int primary key AUTO_INCREMENT,
Imie varchar(25),
Nazwisko varchar (25),
Ulica varchar(25),
Kod varchar(6),
Miasto varchar(25),
PESEL varchar(16),
UNIQUE(PESEL) 
);

create table Wypozyczenia
(
Nr_wyp int primary key AUTO_INCREMENT,
Nr_kl int,
FOREIGN KEY (Nr_kl) references Klienci (Nr_kl),
Nr_ks int,
FOREIGN KEY (Nr_ks) references Ksiazki (Nr_ks),
Data_wyp datetime default CURRENT_DATE(),
Data_odd datetime 
);
